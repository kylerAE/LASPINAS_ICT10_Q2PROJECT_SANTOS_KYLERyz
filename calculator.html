def calculate_gwa(): # Import Element to grab HTML values from js import document # Get Names fname = Element("first_name").element.value lname = Element("last_name").element.value # Get Grades (Handle empty inputs gracefully) try: g1 = float(Element("grade_science").element.value) g2 = float(Element("grade_math").element.value) g3 = float(Element("grade_english").element.value) g4 = float(Element("grade_filipino").element.value) g5 = float(Element("grade_ict").element.value) g6 = float(Element("grade_pe").element.value) # Calculate Average average = (g1 + g2 + g3 + g4 + g5 + g6) / 6 # Update Display Element("result-area").element.style.display = "block" Element("student-name-display").element.innerText = f"Student: {fname} {lname}" Element("gwa-display").element.innerText = f"{average:.2f}" # Simple remark if average >= 90: Element("remark-display").element.innerText = "Ranking: Pirate King Potential!" elif average >= 75: Element("remark-display").element.innerText = "Ranking: Solid Crewmate." else: Element("remark-display").element.innerText = "Ranking: Swab the decks (Failed)." except ValueError: Element("result-area").element.style.display = "block" Element("student-name-display").element.innerText = "Error!" Element("gwa-display").element.innerText = "Please enter numbers."
